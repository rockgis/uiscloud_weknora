## 1. 프로젝트 개요

WeKnora는 문서 이해 및 시맨틱 검색을 위한 엔터프라이즈급 **LLM 기반 RAG (Retrieval-Augmented Generation) 프레임워크**입니다.

### 핵심 기능

- **멀티모달 문서 처리**: PDF, DOCX, XLSX, 이미지 등 다양한 포맷 지원
- **하이브리드 검색**: BM25 키워드 검색 + 벡터 유사도 검색 + 지식 그래프 결합
- **에이전트 모드**: ReACT 패턴 기반 멀티 도구 오케스트레이션
- **멀티 테넌시**: 조직/사용자별 격리된 지식 베이스 관리
- **실시간 스트리밍**: SSE 기반 토큰 단위 응답 스트리밍
- **MCP 통합**: Model Context Protocol을 통한 확장 가능한 도구 시스템

### 핸들러-서비스-리포지토리 패턴

```go
// 핸들러 (HTTP 요청 처리)
type SessionHandler struct {
    sessionService *service.SessionService
}

func (h *SessionHandler) CreateSession(c *gin.Context) {
    var req types.CreateSessionRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        // 에러 처리
    }

    session, err := h.sessionService.CreateSession(ctx, req)
    // 응답 반환
}

// 서비스 (비즈니스 로직)
type SessionService struct {
    sessionRepo    repository.SessionRepository
    knowledgeRepo  repository.KnowledgeBaseRepository
    eventBus       *event.EventBus
}

func (s *SessionService) CreateSession(ctx context.Context, req types.CreateSessionRequest) (*types.Session, error) {
    // 비즈니스 로직
    session := &types.Session{...}

    if err := s.sessionRepo.Create(ctx, session); err != nil {
        return nil, err
    }

    s.eventBus.Publish(event.SessionCreated, session)
    return session, nil
}

// 리포지토리 (데이터 접근)
type SessionRepository interface {
    Create(ctx context.Context, session *types.Session) error
    GetByID(ctx context.Context, id string) (*types.Session, error)
    Update(ctx context.Context, session *types.Session) error
    Delete(ctx context.Context, id string) error
}
```

```typescript
export const useAuthStore = defineStore('auth', {
  state: () => ({
    user: null as User | null,

  actions: {
```

### gRPC 서비스 정의

```protobuf
service DocReader {
  // 문서 파싱 및 청킹
  rpc ParseDocument(ParseRequest) returns (ParseResponse);

  // 스트리밍 파싱 (대용량 문서)
  rpc ParseDocumentStream(ParseRequest) returns (stream Chunk);

  // 헬스 체크
  rpc HealthCheck(Empty) returns (HealthResponse);
}

message Chunk {
```

## 7. RAG 파이프라인

### 파이프라인 아키텍처

`internal/application/service/chat_pipline/` 디렉토리에서 이벤트 기반 플러그인 시스템으로 구현됩니다.

```
┌─────────────────────────────────────────────────────────────────┐
│                      Chat Pipeline Manager                       │
│  ┌───────────────────────────────────────────────────────────┐  │
│  │                     Plugin Chain                           │  │
│  │  ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐          │  │
│  │  │ Rewrite │→│ Search  │→│ Rerank  │→│  Merge  │→ ...     │  │
│  │  └─────────┘ └─────────┘ └─────────┘ └─────────┘          │  │
│  └───────────────────────────────────────────────────────────┘  │
│                              │                                   │
│                              ▼                                   │
│  ┌───────────────────────────────────────────────────────────┐  │
│  │                      Event Bus                             │  │
│  │  OnQuery → OnRewrite → OnSearch → OnRerank → OnComplete   │  │
│  └───────────────────────────────────────────────────────────┘  │
└─────────────────────────────────────────────────────────────────┘
```

### 주요 플러그인

#### 1. 쿼리 리라이팅 (`rewrite.go`)

```go
// 대화 맥락을 반영하여 쿼리 재작성
type RewritePlugin struct {
    llmService LLMService
}

func (p *RewritePlugin) Execute(ctx *PipelineContext) error {
    // 이전 대화 내역 가져오기
    history := ctx.GetConversationHistory()

    // LLM으로 쿼리 재작성
    rewrittenQuery := p.llmService.Rewrite(ctx.Query, history)
    ctx.RewrittenQuery = rewrittenQuery

    return nil
}
```

#### 2. 하이브리드 검색 (`search.go`)

```go
type SearchPlugin struct {
    retrieverRegistry *retriever.Registry
}

func (p *SearchPlugin) Execute(ctx *PipelineContext) error {
    results := []types.Chunk{}

    // BM25 키워드 검색
    keywordResults := p.retrieverRegistry.Search(
        ctx, retriever.TypeKeyword, ctx.Query)

    // 벡터 유사도 검색
    vectorResults := p.retrieverRegistry.Search(
        ctx, retriever.TypeVector, ctx.Query)

    // 결과 병합
    ctx.SearchResults = merge(keywordResults, vectorResults)
    return nil
}
```

#### 3. 리랭킹 (`rerank.go`)

```go
type RerankPlugin struct {
    rerankModel RerankModel
}

func (p *RerankPlugin) Execute(ctx *PipelineContext) error {
    // Rerank 모델로 관련도 재정렬
    scored := p.rerankModel.Rerank(ctx.Query, ctx.SearchResults)

    // 상위 K개 선택
    ctx.SearchResults = scored[:ctx.TopK]
    return nil
}
```

#### 4. LLM 스트리밍 (`chat_completion_stream.go`)

```go
type StreamPlugin struct {
    llmService LLMService
    streamMgr  *stream.Manager
}

func (p *StreamPlugin) Execute(ctx *PipelineContext) error {
    // 컨텍스트 포맷팅
    prompt := formatPrompt(ctx.Query, ctx.SearchResults)

    // 스트리밍 응답
    stream := p.llmService.StreamComplete(prompt)

    for token := range stream {
        p.streamMgr.Send(ctx.SessionID, token)
    }

    return nil
}
```

```sql
CREATE TABLE knowledge (
    id UUID PRIMARY KEY,
    knowledge_base_id UUID REFERENCES knowledge_bases(id),
    name VARCHAR(255),
    source_type VARCHAR(50),  -- 'file' | 'url' | 'text'
    source_url TEXT,
    file_id VARCHAR(255),
    status VARCHAR(50),  -- 'pending' | 'processing' | 'completed' | 'failed'
    metadata JSONB,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
```

```sql
CREATE TABLE chunks (
    id UUID PRIMARY KEY,

    seq INTEGER,
    start_pos INTEGER,
    end_pos INTEGER,
    is_enabled BOOLEAN DEFAULT true,
    metadata JSONB,
    created_at TIMESTAMP
);
```
